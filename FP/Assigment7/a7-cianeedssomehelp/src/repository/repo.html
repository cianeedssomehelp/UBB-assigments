 8class BookMemoryRepo:
 9    def __init__(self):
10        self._data = []
11        self._history = [deepcopy(self._data)]
12
13    def get_book(self, isbn: str):
14        """
15        This function will return a book object.
16        :param isbn: the unique code by which the book is identified.
17        :return: the book.
18        """
19        for book in self._data:
20            if book.get_Isbn() == isbn:
21                return book
22
23    def add_book(self, book: Book):
24        """
25        This function will add a book to the repository.
26        :param book: the object we want to add.
27        :return: the new list of objects
28        """
29        for existing_book in self._data:
30            if existing_book.get_Isbn() == book.get_Isbn():
31                raise RepositoryError(f"Book with ISBN {book.get_Isbn()} already exists.")
32        self._data.append(book)
33
34    def remove_book(self, book: Book):
35        """
36        This function will remove a book from the repository.
37        :param book: the object we want to remove.
38        :return: the new list of books
39        """
40        self._data.remove(book)
41
42    def add_to_history(self):
43        """
44        This function will add a book to the history.
45        :return: the history of books
46        """
47        self._history.append(deepcopy(self._data))
48
49    def undo(self):
50        """
51        this function undoes any operations we make
52        :return: the last state of the repository
53        """
54        if len(self._history) > 1:
55            self._history.pop()
56            self._data = deepcopy(self._history[-1])
57        else:
58            raise RepositoryError("No operations to undo.")
59    def get_all_books(self):
60        """
61        This function will get the list of all books.
62        :return: the list of books
63        """
64        return self._data
65
66    def __len__(self):
67        return len(self._data)
68
69    def clear(self):
70        """
71        This function will clear the repository.
72        :return: fresh clean repository
73        """
74        self._data.clear()

 5class RepositoryError(Exception):
 6    pass

 8class BookTextFileRepo(BookMemoryRepo):
 9    def __init__(self):
10        self.__filename = "books.txt"
11        self._data = []
12        self._history = []
13        if self.load_file():
14            self._history.append(deepcopy(self._data))
15        else:
16            super().__init__()
17
18    def load_file(self):
19        """
20        Loads the books into a text file.
21        """
22        try:
23            fin = open(self.__filename, "rt")
24            lines = fin.readlines()
25            fin.close()
26        except IOError:
27            return False
28        for line in lines:
29            if line == "":
30                continue
31            line = line.strip()
32            line = line.split(",")
33            book = Book(line[0], line[1], line[2])
34            self._data.append(book)
35        if len(super().get_all_books()) == 0:
36            return False
37        return True
38
39    def __saveFile(self):
40        """
41        Saves the books into a text file.
42        """
43        try:
44            fout = open(self.__filename, "wt")
45            books = super().get_all_books()
46            for book in books:
47                book_string = str(book.get_Isbn()) + "," + str(book.get_Author()) + "," + str(book.get_Title()) + '\n'
48                fout.write(book_string)
49        except IOError:
50            raise RepositoryError("Save file not found.")
51
52    def add_book(self, book):
53        """
54        This function adds a book to the repository.
55        :param book: our book
56        :return: the new list of books
57        """
58        super().add_book(book)
59        self.__saveFile()
